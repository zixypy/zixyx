local function addUICorner(instance)
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 10)
    corner.Parent = instance
end

addUICorner(frame)
addUICorner(keyTextBox)
addUICorner(verifyButton)
addUICorner(getKeyButton)

-- Button Logic
verifyButton.MouseButton1Click:Connect(function()
    local key = keyTextBox.Text
    local status = api.check_key(key)

    if (status.code == "KEY_VALID") then
         screenGui:Destroy()
        script_key = keyTextBox.Text
        api.load_script();
        return
    elseif (status.code == "KEY_HWID_LOCKED") then
        keyTextBox.Text = ""
        keyTextBox.PlaceholderText = "Key HWID Locked"
        return

    else
        keyTextBox.Text = ""
        keyTextBox.PlaceholderText = "Invalid Key"
        return
    end
end)

getKeyButton.MouseButton1Click:Connect(function()
    setclipboard("https://ads.luarmor.net/get_key?for=Zixy_X-ZwPSxFJIbjyo")
    keyTextBox.PlaceholderText = "Copied key system!"
end)
